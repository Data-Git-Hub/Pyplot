letter_freq = dict(Counter(TEXT))

# make a bar plot with an appropriate title, correct xtick labels, and labeled axes

# Bar Plot 1: Character frequencies sorted alphabetically
sorted_alpha = dict(sorted(letter_freq.items(), key=lambda x: x[0]))

plt.figure(figsize=(14, 5))
plt.bar(sorted_alpha.keys(), sorted_alpha.values(), color='steelblue')
plt.title("Character Frequency Distribution (Alphabetical Order)")
plt.xlabel("Character")
plt.ylabel("Frequency")
plt.xticks(rotation=90)
plt.tight_layout()
plt.show()

# Bar Plot 2: Sorted by frequency (descending), then alphabetically
sorted_freq = sorted(letter_freq.items(), key=lambda x: (-x[1], x[0]))

plt.figure(figsize=(14, 5))

plt.title("Character Frequency Distribution (Sorted by Frequency & Alphabetical Order)")
plt.xlabel("Character")
plt.ylabel("Frequency")
plt.xticks(rotation=90)

plt.show()

Section 2. Visualizing a Sequence of Random Prime Numbers

Generate a sequence of at least 5 random numbers (integers are fine), and visualize the data using both scatter and plot on different axes. You may use the index of the value as its x value in the plot.

In[20]:
from random import sample

# Define all prime numbers between 2 and 199
def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5)+1):
        if n % i == 0:
            return False
    return True

primes = [n for n in range(2, 200) if is_prime(n)]

# Select 13 unique random primes
random_primes = sample(primes, 13)

# Print the selected primes
print("Randomly selected prime numbers (unordered):")
print(random_primes)

# Create x values as indices
x_values = list(range(len(random_primes)))

# Scatter Plot
plt.figure(figsize=(10, 4))
plt.scatter(x_values, random_primes, color='mediumseagreen')
plt.title("Scatter Plot of Random Prime Numbers")
plt.xlabel("Index")
plt.ylabel("Prime Value")
plt.grid(True)
plt.tight_layout()
plt.show()

# Line Plot
plt.figure(figsize=(10, 4))
plt.plot(x_values, random_primes, marker='o', linestyle='-', color='darkred')
plt.title("Line Plot of Random Prime Numbers")
plt.xlabel("Index")
plt.ylabel("Prime Value")
plt.grid(True)
plt.tight_layout()
plt.show()
